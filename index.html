PRR Process Flowchart

flowchart TB
    %% VAULTPRR™ — Massachusetts Public Records Requests
    %% Canonical CASE-Based Architecture v1.0

    CASE["PRR CASE Workspace"]:::core

    CASE --> TIMERS["Statutory Timers\nT10 | T20 | T25 | T90"]
    CASE --> ASSETS["Assets\n(All Documents)"]
    CASE --> LOG["Audit Log"]
    CASE --> PROCESS["Process Execution"]

    PROCESS --> START([S000: PRR Received])
    START --> INTAKE["S100: Intake\nFormStep"]
    INTAKE --> CALC["S200: Compute Timers\nSystemEvent"]
    CALC --> ASSESS["S300: RAO Initial Assessment\nReviewStep | KEEPER"]

    ASSESS -->|Clarification Needed| CLARIFY["S310: Request Clarification"]
    CLARIFY --> WAIT["S320: Wait for Response\nWaitState | Clock Tolled"]
    WAIT -->|Response Received| TOLL["Toll & Recompute Timers"]
    TOLL --> ASSESS
    WAIT -->|30 days no response| TIMEOUT["S991: Timeout\nTerminal"]

    ASSESS -->|No Responsive Records| NOREC["S910: No Records Letter\nDocumentStep | LOCKED"]
    NOREC --> CLOSE_NO["Close CASE\nOutcome: No Records"]

    ASSESS --> FORECAST{"S330: Forecast > T25?"}
    FORECAST -->|No| GATHER
    FORECAST -->|Yes + Agreement| AGREED["S340: Extension Agreement\nLOCKED"]
    FORECAST -->|Yes + No Agreement| PETITION["S350: Petition to Supervisor\nLOCKED"]
    AGREED --> GATHER
    PETITION --> GATHER

    ASSESS -->|Proceed| GATHER["S400: Gather / Search Records"]
    GATHER --> EXEMPT["S500: Exemption Review\nReviewStep | KEEPER"]

    EXEMPT -->|All Exempt| DENY["S920: Withholding Letter\nLOCKED"]
    DENY --> CLOSE_DENY["Close CASE\nOutcome: Withheld"]

    EXEMPT -->|Partial or None| REDACT{"Redactions Needed?"}
    REDACT -->|Yes| APPLY["S510: Apply Redactions"]
    APPLY --> LEGAL["S520: Counsel Review\nGateStep"]
    LEGAL -->|Revise| APPLY
    LEGAL -->|Approved| PACKAGE["S600: Response Package\nLOCKED"]
    REDACT -->|No| PACKAGE

    PACKAGE --> T10CHECK{"S700: T10 Met?"}
    T10CHECK -->|No| FEEWAIVE["FeesAllowed = false"]
    T10CHECK -->|Yes| FEES{"Fees Required?"}

    FEES -->|Yes| ESTIMATE["Itemized Fee Estimate"]
    ESTIMATE --> INVOICE["S710: Invoice"]
    INVOICE --> PAYMENT{"S720: Payment Received?"}
    PAYMENT -->|No - 30 days| CLOSE_NONPAY["S930: Close CASE\nOutcome: Non-Payment"]
    PAYMENT -->|Yes| DELIVER["S800: Deliver Records"]

    FEES -->|No| DELIVER
    FEEWAIVE --> DELIVER

    DELIVER --> CLOSE_OK["S900: Close CASE\nOutcome: Delivered"]
